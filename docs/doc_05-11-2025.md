**PDF-to-Image Conversion Function**
```python
def pdf_page_to_image(pdf_path: str, page_num: int = 0, max_pages: int = 3):
    """Convert PDF pages to images. Returns list of image file paths."""
    if not PDF2IMAGE_AVAILABLE:
        print(f"[DEBUG] pdf2image not available.", flush=True)
        return []
    
    if not os.path.exists(pdf_path):
        print(f"[DEBUG] PDF file not found: {pdf_path}", flush=True)
        return []
    
    try:
        reader = PdfReader(pdf_path)
        total_pages = len(reader.pages)
        last_page = min(page_num + max_pages, total_pages)
        
        convert_kwargs = {
            "first_page": page_num + 1,
            "last_page": last_page,
            "dpi": 150
        }
        
        if POPPLER_PATH:
            convert_kwargs["poppler_path"] = POPPLER_PATH
        
        images = convert_from_path(pdf_path, **convert_kwargs)
        
        image_paths = []
        pdf_name = Path(pdf_path).stem
        timestamp = datetime.now().strftime('%Y%m%d_%H%M%S')
        
        for i, image in enumerate(images):
            image_path = os.path.join(
                assignment_images_directory,
                f"{pdf_name}_page_{i+1}_{timestamp}.png"
            )
            image.save(image_path, 'PNG')
            image_paths.append(image_path)
        
        return image_paths
    except Exception as e:
        print(f"[ERROR] PDF conversion failed: {e}", flush=True)
        return []
```

**Enhanced Assignment File Discovery**
```python
def get_relevant_assignment_files(query: str, max_files: int = 3):
    """Find relevant assignment files from the KB based on the query."""
    context = rag_search(query, k=12)  # Increased from 8 to 12
    print(f"[DEBUG] RAG context: {context[:300]}...", flush=True)
    
    # Extract file sources from context
    file_sources = set()
    for line in context.split('\n'):
        if line.startswith('[') and ']' in line:
            source = line.split(']')[0].strip('[')
            file_sources.add(source)
    
    print(f"[DEBUG] Found file sources from RAG: {file_sources}", flush=True)
    
    # Find actual PDF files in kb directory
    pdf_files = []
    kb_base = "kb"
    
    for file_source in list(file_sources)[:max_files * 2]:
        possible_paths = [
            os.path.join(kb_base, file_source),
            os.path.join(kb_base, os.path.basename(file_source)),
            file_source,
        ]
        for path in possible_paths:
            if os.path.exists(path) and path.lower().endswith('.pdf'):
                pdf_files.append(path)
                print(f"[DEBUG] Found PDF from RAG: {path}", flush=True)
                break
    
    # Fallback: prioritize assignment/homework PDFs
    if not pdf_files:
        print(f"[DEBUG] No PDFs found from RAG, using fallback...", flush=True)
        all_pdfs = list([f for f in iter_kb_files() if f.lower().endswith('.pdf')])
        print(f"[DEBUG] Total PDFs found in KB: {len(all_pdfs)}", flush=True)
        
        if all_pdfs:
            assignment_pdfs = [f for f in all_pdfs if any(keyword in f.lower() 
                for keyword in ['hw', 'assignment', 'project', 'paltsokas', 'dama'])]
            pdf_files = assignment_pdfs[:max_files] if assignment_pdfs else all_pdfs[:max_files]
            print(f"[DEBUG] Using prioritized PDFs: {pdf_files}", flush=True)
    
    return pdf_files
```

**Chat with Images Integration**
```python
def chat_with_images(self, message, history):
    """Chat function that can return both text and images for assignment questions."""
    message_lower = message.lower()
    is_assignment_question = (
        "tell me in detail about a project or assignment" in message_lower or
        "screenshot" in message_lower or
        "graph" in message_lower or
        "visualization" in message_lower or
        ("provide" in message_lower and "image" in message_lower)
    )
    
    text_response = self.chat(message, history)
    
    if is_assignment_question:
        print(f"[DEBUG] Attempting to get assignment images...", flush=True)
        
        pdf_files = get_relevant_assignment_files(message)
        all_image_paths = []
        
        for pdf_file in pdf_files:
            image_paths = pdf_page_to_image(pdf_file)
            all_image_paths.extend(image_paths)
        
        if all_image_paths:
            return {
                "text": text_response + "\n\nüìé **Relevant Assignment Pages:**",
                "files": all_image_paths[:3]
            }
    
    return text_response
```

## 15) Lessons Learned

1. **Gradio Image Rendering**: ChatInterface doesn't easily support embedded images - requires specific format or alternative components
2. **Base64 Encoding**: Not suitable for large images in chat interfaces - creates unreadable walls of text
3. **Poppler Configuration**: Windows requires explicit path configuration - Linux/Mac is simpler
4. **RAG File Discovery**: Need robust fallback logic when semantic search doesn't find exact PDF matches
5. **Python Environment**: Always verify packages are installed in correct venv, not global Python
6. **Indentation Errors**: Small indentation issues can break entire app - syntax validation is critical
7. **Git Best Practices**: Large binary files (Poppler, screenshots) should be gitignored
